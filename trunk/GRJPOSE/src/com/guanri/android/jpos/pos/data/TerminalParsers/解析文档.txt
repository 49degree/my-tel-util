各种交易报文的解析文档
解析TTransaction类


***签到*****
判别码: 
	if (Transaction.TransCode().GetAsInteger() == 1) {
	}

读取:
	Transaction.SerialNumber().GetAsInteger();						//流水号
	Transaction.Year().GetAsString();								//年, YYYY
	Transaction.Date().GetAsString();								//月日, MMDD
	Transaction.Time().GetAsString();								//时分秒, hhmmss
	Transaction.ProcessList.MerchantID().GetAsString();				//商户号
	Transaction.ProcessList.TerminalID().GetAsString();				//终端号
	Transaction.ProcessList.UserID().GetAsString();					//操作员
	
回送:	
	Transaction.TransCode().GetAsInteger(1);						//交易代码
	Transaction.SerialNumber().SetAsInteger();						//流水号
	Transaction.Year().SetAsString();								//年, YYYY
	Transaction.Date().SetAsString();								//月日, MMDD
	Transaction.Time().SetAsString();								//时分秒, hhmmss
	
	Transaction.ProcessList.Response().SetAsString();				//应答
	Transaction.ProcessList.TerminalName().SetAsString();			//商户名称
	
应答格式:	
"00" + "签到成功"
xx + 错误提示信息


***查询订单*****
判别码: 
	if (Transaction.TransCode().GetAsInteger() == 600) {
	}

读取:
	Transaction.SerialNumber().GetAsInteger();						//流水号
	Transaction.Year().GetAsString();								//年, YYYY
	Transaction.Date().GetAsString();								//月日, MMDD
	Transaction.Time().GetAsString();								//时分秒, hhmmss
	Transaction.ProcessList.MerchantID().GetAsString();				//商户号
	Transaction.ProcessList.TerminalID().GetAsString();				//终端号
	Transaction.ProcessList.UserID().GetAsString();					//操作员
	
	Transaction.ProcessList.OrderNumber().GetAsString();			//订单编号
	
回送:	
	Transaction.TransCode().GetAsInteger(600);						//交易代码
	Transaction.SerialNumber().SetAsInteger();						//流水号
	Transaction.Year().SetAsString();								//年, YYYY
	Transaction.Date().SetAsString();								//月日, MMDD
	Transaction.Time().SetAsString();								//时分秒, hhmmss
	
	Transaction.ProcessList.Response().SetAsString();				//应答
	Transaction.ProcessList.ReturnOrderNumber().SetAsString();		//回送订单编号
	Transaction.ProcessList.ReturnSaleAmount().SetAsString();		//回送消费金额
	Transaction.ProcessList.ReturnDisplayMessage().SetAsString();	//回送显示信息
	
应答格式:	
"00" + 提示信息(订单编号，订单金额等等, 暂定使用57域的数据)
xx + 错误提示信息



***查询后消费*****
判别码: 
	if (Transaction.TransCode().GetAsInteger() == 601) {
	}

读取:
	Transaction.SerialNumber().GetAsInteger();						//流水号
	Transaction.Year().GetAsString();								//年, YYYY
	Transaction.Date().GetAsString();								//月日, MMDD
	Transaction.Time().GetAsString();								//时分秒, hhmmss
	Transaction.ProcessList.MerchantID().GetAsString();				//商户号
	Transaction.ProcessList.TerminalID().GetAsString();				//终端号
	Transaction.ProcessList.UserID().GetAsString();					//操作员
	
	Transaction.ProcessList.ReturnOrderNumber().GetAsString();		//订单编号
	Transaction.ProcessList.ReturnSaleAmount().GetAsString();		//消费金额
	Transaction.ProcessList.ReturnDisplayMessage().GetAsString();	//显示信息
	
	Transaction.ProcessList.GetIsExistTrackData();					//是否存在磁道数据
	Transaction.ProcessList.GetIsExistPINData();					//是否存在密码数据
	Transaction.ProcessList.GetTrack2Data();						//2磁道数据
	Transaction.ProcessList.GetTrack3Data();						//3磁道数据
	Transaction.ProcessList.GetPAN();								//主帐号
	Transaction.ProcessList.PINData().GetData;						//密码
	
回送:	
	Transaction.TransCode().GetAsInteger(601);						//交易代码
	Transaction.SerialNumber().SetAsInteger();						//流水号
	Transaction.Year().SetAsString();								//年, YYYY
	Transaction.Date().SetAsString();								//月日, MMDD
	Transaction.Time().SetAsString();								//时分秒, hhmmss
	
	Transaction.ProcessList.Response().SetAsString();				//应答

	
应答格式:	
"00" + "交易成功"
xx + 错误提示信息
